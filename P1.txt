/**
 * Definiciones básicas.
 */
Program ::= header_program block;

block ::= start_block local_var_dec sub_progs sentences end_block ;

sub_progs ::= sub_progs sub_prog | ;

sub_prog ::= header_subprogram block;

local_var_dec ::= LIMIT_INI local_var LIMIT_END | ;

header_program ::= MAIN;

start_block ::= BEGIN;

end_block ::= END;

local_var ::= local_var var_body | 
            var_body;

var_body ::= type list_id SEMICOLON;

header_subprogram ::= type id PL parameters PR | 
            type id PL PR;

sentences ::= sentences sentence | 
            sentence;

sentence ::= block | 
            sentence_alloc | 
            sentence_if | 
            sentence_else | 
            sentence_while | 
            sentence_input | 
            sentence_output | 
            sentence_return | 
            sentence_do_until | 
            sentence_forward | 
            sentence_back | 
            sentence_start_cursor;

sentence_alloc ::= id ALLOC expr SEMICOLON;

sentence_if ::= IF PL expr PR sentence | 
            IF PL expr PR sentence sentence_else;

sentence_else ::= ELSE sentence;

sentence_while ::= WHILE PL expr PR sentence;

sentence_input ::= INPUT input_list SEMICOLON;

sentence_output ::= OUTPUT exprs_cad SEMICOLON;

sentence_return ::= RETURN expr SEMICOLON;

sentence_do_until ::= DO sentence UNTIL PL expr PR SEMICOLON;

sentence_forward ::= expr OP_LIST;

sentence_back ::= expr OP_LIST;

sentence_start_cursor ::= OP_START_CURSOR expr;

exprs ::= exprs COMMA expr | 
            expr;

expr ::= PL expr PR | 
          OP_UNIT expr | 
          expr OP_BIN expr | 
          expr OP_BIN expr OP_BIN expr | 
          id | 
          const | 
          function_call | 
          const_list;

exprs_cad ::= exprs_cad COMMA expr_cad | 
            expr_cad;

expr_cad ::= expr | 
            cad;

/**
 * Definiciones auxiliares.
 */

parameters ::= parameters COMMA type id | 
            type id;

list_id ::= list_id COMMA id | 
            id;

input_list ::= input_list COMMA id | 
            id;

function_call ::= id PL exprs PR;

const_list ::= BEGIN_LIST elements_list END_LIST;

elements_list ::= elements_list COMMA const | 
            const;

ALLOC ::= '=' ;

cad ::= cad letter | 
            cad digit | 
            letter | 
            digit | 
            _cad;

id ::= letter 
            | id [ letter | digit ];

const ::= c_int | 
            c_float | 
            c_char | 
            c_bool;

type ::= int | 
            float | 
            char | 
            bool | 
            list type;

c_bool ::= const bool;

c_char ::= const char;

c_float ::= const float;

c_int ::= const int;

list ::= 'list of';
 
bool ::= 'true' | 
            'false';

char ::= digit | 
            letter;

float ::= int '.' [digit];

int ::= ['-'] digit | 
            digit | 
            int digit;

letter = a | b | c | ... | z | A | B | ... | Z;

digit = 0 | 1 | 2 | 3 | 4 | ... | 9;

PL ::= '(' ;

PR ::= ')' ;

COMMA ::= ',' ;

SEMICOLON ::= ';' ;

BEGIN ::= '{' ;

END ::= '}' ;

BEGIN_LIST ::= '[' ;

END_LIST ::= ']' ;

OP_BIN ::= / | % | > | < | >= | <= | == | != | || | && | ^ | + | - | * | ** | @ | -- ;

OP_UNIT ::= ! | + | - | # | ? | ++ | -- ;

OP_LIST ::= << | >> | $;
